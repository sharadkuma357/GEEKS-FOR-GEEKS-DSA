class Solution {

  public:
    vector<int> help_classmate(vector<int> arr, int n) {
        // Your code goes here
        vector<int> result(n, -1);
        stack<int> s;
        
        for (int i = n - 1; i >= 0; --i) {
            while (!s.empty() && s.top() >= arr[i]) {
                s.pop();
            }
            if (!s.empty()) {
                result[i] = s.top();
            }
            s.push(arr[i]);
        }
        
        return result;
    }
};
