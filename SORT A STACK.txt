class Solution {
  public:
    vector<int> leftSmaller(vector<int> arr) {
        // code here
         vector<int> res(arr.size(), -1);
    stack<int> stk;
    
    for (int i = 0; i < arr.size(); i++) {
        
        while (!stk.empty() && stk.top() >= arr[i]) {
            stk.pop();
        }
        
        if (!stk.empty()) {
            res[i] = stk.top();
        }
        
        stk.push(arr[i]);
    }
    
    return res;
    }
};
